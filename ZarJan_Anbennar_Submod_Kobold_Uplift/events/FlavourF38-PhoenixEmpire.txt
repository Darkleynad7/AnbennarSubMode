#The Length of the Divenhal
country_event = {
	id = phoenix_empire.6
	title = phoenix_empire.6.t
	desc = phoenix_empire.6.d
	picture = SHIP_SAILING_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				any_country = {
					capital_scope = { OR = { continent = north_america continent = south_america } }
					religion = bulwari_sun_cult
					any_owned_province = {
						has_port = yes
						NOT = { development = 15 }
					}
				}
			}
			random_country = {
				limit = {
					capital_scope = { OR = { continent = north_america continent = south_america } }
					religion = bulwari_sun_cult
					any_owned_province = {
						has_port = yes
					}
				}
				random_owned_province = {
					limit = { has_port = yes NOT = { development = 15 } }
					save_event_target_as = phoenix_empire_exclave
				}
			}
		}
		else_if = {
			limit = {
				any_country = {
					capital_scope = { OR = { continent = north_america continent = south_america } }
					any_owned_province = {
						has_port = yes
						NOT = { development = 15 }
					}
				}
			}
			random_country = {
				limit = {
					capital_scope = { OR = { continent = north_america continent = south_america } }
					any_owned_province = {
						has_port = yes
					}
				}
				random_owned_province = {
					limit = { has_port = yes NOT = { development = 15 } }
					save_event_target_as = phoenix_empire_exclave
				}
			}
		}
		else = {
			random_country = {
				limit = {
					capital_scope = { OR = { continent = north_america continent = south_america } }
					any_owned_province = {
						has_port = yes
					}
				}
				random_owned_province = {
					limit = { has_port = yes }
					save_event_target_as = phoenix_empire_exclave
				}
			}
		}
	}
	
	option = {
		name = phoenix_empire.6.a
		ai_chance = {
			factor = 1
		}
		
		event_target:phoenix_empire_exclave = {
			change_religion = bulwari_sun_cult
			if = {
				limit = { development = 15 }
				add_elven_minority_size_effect = yes
			}
			else = {
				change_culture = sun_elf
			}
			add_base_tax = 5
			add_base_production = 5
			add_base_manpower = 5
			add_core = ROOT
		}
	}
}
